export function hasDecimals(separator: string, decimalPlaces: string) {
  return new RegExp(`^\\d+\\${separator}\\d{${decimalPlaces}}$`, 'g');
}
 
export function hasProtocol(url: string) {
  return /^[a-z]*:\/\//.test(url);
}
